764template <class Type>void indexx(vector<Type>&arr, vector<Type>&indx)
765{
766 const int M=7,NSTACK=50;
767 ULONG n = indx.size();
768 long ir;
769 long i,j,k;
770 ULONG indxt;
771 long jstack=-1;
772 long l=0;
773 Type a;
774 int istack[NSTACK];
775
776 ir=n-1;
777 for (j=0;j<n;j++) indx[j]=j;
778 for (;;) {
779 if (ir-l < M)
780 {
781 for (j=l+1;j<=ir;j++)
782 {
783 indxt=indx[j];
784 a=arr[indxt];
785 for (i=j-1;i>=l;i--)
786 {
787 if (arr[indx[i]] <= a) break;
788 indx[i+1]=indx[i];
789 }
790 indx[i+1]=indxt;
791 }
792 if (jstack < 0) break;
793 ir=istack[jstack--];
794 l=istack[jstack--];
795 }
796 else
797 {
798 k=(l+ir) >> 1;
799 SWAP_L<Type>(indx[k],indx[l+1]);
800 if (arr[indx[l]] > arr[indx[ir]]) {
801 SWAP_L<Type>(indx[l],indx[ir]);
802 }
803 if (arr[indx[l+1]] > arr[indx[ir]]) {
804 SWAP_L<Type>(indx[l+1],indx[ir]);
805 }
806 if (arr[indx[l]] > arr[indx[l+1]]) {
807 SWAP_L<Type>(indx[l],indx[l+1]);
808 }
809 i=l+1;
810 j=ir;
811 indxt=indx[l+1];
812 a=arr[indxt];
813 for (;;) {
814 do i++; while (arr[indx[i]] < a);
815 do j--; while (arr[indx[j]] > a);
816 if (j < i) break;
817 SWAP_L<Type>(indx[i],indx[j]);
818 }
819 indx[l+1]=indx[j];
820 indx[j]=indxt;
821 jstack += 2;
822 if (jstack >= NSTACK) {
823 cerr << "NSTACK too small in indexx." << endl;
824 exit(1);
825 }
826 if (ir-i+1 >= j-l) {
827 istack[jstack]=ir;
828 istack[jstack-1]=i;
829 ir=j-1;
830 }
831 else {
832 istack[jstack]=j-1;
833 istack[jstack-1]=l;
834 l=i;
835 }
836 }
837 }
838}
