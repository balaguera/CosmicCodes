671template<typename Type> Type get_mean_occupation_number(real_prec max, const vector<Type>&in)
672{
673 if(in.size()==0)
674 cerr<<"Error. Empty vector"<<endl;
675 ULONG Ntot=0;
676 int Nd=static_cast<int>(max);
677 vector<int>dist(Nd+1,0);
678 real_prec delta=1;
679 for(int i=0;i<in.size();++i)
680 dist[get_bin(in[i],0,Nd,delta,true)]++;
681 int mean=0;
682 Ntot=0;
683 for(int i=0;i<=Nd;++i)
684 {
685 mean+=i*dist[i];
686 Ntot+=dist[i];
687 }
688 mean/=static_cast<real_prec>(Ntot);
689 return static_cast<int>(mean);
690}
