37template <int sz, typename T1, typename T2=char, typename T3=char>
38 struct sizeChooser
39 {
40 typedef typename sizeChooserHelper2
41 <typename sizeChooserHelper<T1,sizeof(T1)==sz>::TYPE,
42 typename sizeChooserHelper<T2,sizeof(T2)==sz>::TYPE,
43 typename sizeChooserHelper<T3,sizeof(T3)==sz>::TYPE >::TYPE TYPE;
44 };
